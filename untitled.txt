<script>
let lat, lon;
let prayerTimes = {};
let tasbihCount = localStorage.getItem('tasbih_val') || 0;

// 1. الساعة والوقت
function updateTime() {
    const now = new Date();
    document.getElementById('currentTime').innerText = now.toLocaleTimeString('ar-EG');
    document.getElementById('currentDateAr').innerText = now.toLocaleDateString('ar-EG', {weekday:'long', day:'numeric', month:'long', year:'numeric'});
}
setInterval(updateTime, 1000);
updateTime();

// 2. تحديد الموقع
function getLocation() {
    document.getElementById('locationName').innerText = "جاري التحديد...";
    navigator.geolocation.getCurrentPosition(pos => {
        lat = pos.coords.latitude;
        lon = pos.coords.longitude;
        document.getElementById('locationName').innerText = "تم التحديد ✅";
        fetchPrayers();
        fetchHijri();
        calculateQibla();
    }, () => {
        document.getElementById('locationName').innerText = "تعذر الوصول للموقع (افتراضي: القاهرة)";
        lat = 30.0444; lon = 31.2357;
        fetchPrayers();
        fetchHijri();
        calculateQibla();
    });
}

// 3. جلب المواقيت
async function fetchPrayers() {
    const res = await fetch(`https://api.aladhan.com/v1/timings?latitude=${lat}&longitude=${lon}&method=5`);
    const data = await res.json();
    prayerTimes = data.data.timings;
    
    const pNames = {Fajr:'الفجر', Sunrise:'الشروق', Dhuhr:'الظهر', Asr:'العصر', Maghrib:'المغرب', Isha:'العشاء'};
    let html = '';
    for(let k in pNames) {
        html += `<div class="prayer-card">
                    <div class="prayer-name-ar">${pNames[k]}</div>
                    <div class="prayer-time">${prayerTimes[k]}</div>
                 </div>`;
    }
    document.getElementById('prayersGrid').innerHTML = html;
    document.getElementById('nextPrayerBox').style.display = 'flex';
    updateNextPrayer();
}

// 4. القبلة
function calculateQibla() {
    const mLat = 21.4225, mLon = 39.8262;
    const y = Math.sin(deg2rad(mLon - lon));
    const x = Math.cos(deg2rad(lat)) * Math.tan(deg2rad(mLat)) - Math.sin(deg2rad(lat)) * Math.cos(deg2rad(mLon - lon));
    let bearing = rad2deg(Math.atan2(y, x));
    bearing = (bearing + 360) % 360;
    document.getElementById('qiblaDeg').innerText = Math.round(bearing) + "°";
    document.getElementById('needle').style.transform = `rotate(${bearing}deg)`;
}
function deg2rad(d){ return d * (Math.PI/180); }
function rad2deg(r){ return r * (180/Math.PI); }

// 5. السبحة
document.getElementById('countDisplay').innerText = tasbihCount;
function incrementTasbih() {
    tasbihCount++;
    document.getElementById('countDisplay').innerText = tasbihCount;
    localStorage.setItem('tasbih_val', tasbihCount);
    if(navigator.vibrate) navigator.vibrate(50);
}
function resetTasbih() {
    tasbihCount = 0;
    document.getElementById('countDisplay').innerText = 0;
    localStorage.setItem('tasbih_val', 0);
}

// 6. التاريخ الهجري والمناسبات
async function fetchHijri() {
    const res = await fetch(`https://api.aladhan.com/v1/gToH?date=${new Date().getDate()}-${new Date().getMonth()+1}-${new Date().getFullYear()}`);
    const data = await res.json();
    const h = data.data.hijri;
    document.getElementById('currentDateHijri').innerText = `${h.day} ${h.month.ar} ${h.year} هـ`;
    
    const evs = [
        {n:"رمضان المبارك", d:"01 رمضان"}, {n:"عيد الفطر", d:"01 شوال"},
        {n:"يوم عرفة", d:"09 ذو الحجة"}, {n:"عيد الأضحى", d:"10 ذو الحجة"}
    ];
    document.getElementById('eventsList').innerHTML = evs.map(e => `
        <div style="display:flex; justify-content:space-between; padding:12px 0; border-bottom:1px solid #f4f4f4">
            <span>${e.n}</span>
            <span style="color:var(--gold); font-weight:700">${e.d}</span>
        </div>
    `).join('');
}

// 7. التبويبات
function showTab(id, btn) {
    document.getElementById('tab-qibla').style.display = (id==='qibla'?'block':'none');
    document.getElementById('tab-tasbih').style.display = (id==='tasbih'?'block':'none');
    document.getElementById('tab-events').style.display = (id==='events'?'block':'none');
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
}
